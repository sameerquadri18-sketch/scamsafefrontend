### 1.4 Auth Routes

```
POST /api/auth/register
  Body: {name, email, phone, password}
  → Hash password with bcrypt
  → Create user with is_verified=False
  → Generate 6-digit OTP, expires 10 minutes
  → In TEST_MODE: print OTP to console
  → In production: send via Resend email
  → Return: {success: true, message: "OTP sent to email"}

POST /api/auth/verify-otp
  Body: {email, otp}
  → In TEST_MODE: accept any 6-digit code
  → Check OTP + expiry
  → Set is_verified=True
  → Return: {access_token, user}

POST /api/auth/login
  Body: {email, password}
  → Verify password with bcrypt
  → Require is_verified=True (or skip in TEST_MODE)
  → Return: {access_token, user}
  → Frontend: if onboarding_completed → /app/signals
              else → /onboarding

POST /api/auth/resend-otp
  Body: {email}
  → Generate new OTP, print to console in TEST_MODE

JWT tokens:
  Include in Authorization: Bearer {token}
  All /api/* routes except /auth/* require token
  ProtectedRoute redirects to /login NOT to /
```

def check_watch_zone_limit(current_user):
    limits = PLAN_LIMITS[current_user.plan]
    max_zones = limits["watch_zones"]
    if max_zones != -1 and current_user.watch_zones_count >= max_zones:
        raise HTTPException(403, {

def check_signal_limit(current_user):

def test(name, method, path, body=None,

    global PASS, FAIL
    headers = {"Content-Type": "application/json"}
    if token:
        headers["Authorization"] = f"Bearer {token}"
    try:
        if method == "GET":
            r = httpx.get(BASE + path, headers=headers)
        elif method == "POST":
            r = httpx.post(BASE + path,

        elif method == "PATCH":
            r = httpx.patch(BASE + path,

        status = "PASS" if ok else "FAIL"

### 2.1 SerpApi Service (services/serpapi_service.py)

```python
import httpx
import asyncio
import random
import time

SERPAPI_KEY = settings.SERPAPI_KEY

async def search_platform(query: str, platform: str,
                          num: int = 10) -> list:
    """Search a specific platform via SerpApi."""
    site_map = {
        "reddit":   "site:reddit.com",
        "quora":    "site:quora.com",
        "indiamart":"site:indiamart.com",
        "youtube":  "site:youtube.com",
    }
    site = site_map.get(platform, "")
    full_query = f"{site} {query}" if site else query

    try:
        async with httpx.AsyncClient(timeout=15) as client:
            r = await client.get(
                "https://serpapi.com/search",
                params={
                    "q": full_query,
                    "api_key": SERPAPI_KEY,
                    "num": num,
                    "gl": "in",           # India results
                    "hl": "en",
                    "tbs": "qdr:w",       # last week
                }
            )
            data = r.json()
            results = []
            for item in data.get("organic_results", []):
                results.append({
                    "platform":  platform,
                    "title":     item.get("title", ""),
                    "snippet":   item.get("snippet", ""),
                    "link":      item.get("link", ""),
                    "date":      item.get("date", ""),
                })
            return results
    except Exception as e:
        print(f"[SerpApi] {platform} error: {e}")
        return []


async def search_all_platforms(query: str) -> dict:
    """Search all platforms concurrently."""
    platforms = ["reddit", "quora", "indiamart", "youtube"]
    tasks = [search_platform(query, p) for p in platforms]

    # Small random delay to avoid rate limits
    await asyncio.sleep(random.uniform(0.5, 1.5))

    results = await asyncio.gather(*tasks,
                                   return_exceptions=True)
    combined = {}
    for platform, result in zip(platforms, results):
        if isinstance(result, Exception):
            combined[platform] = []
        else:
            combined[platform] = result
    return combined
```

### 2.2 Signal Scorer (services/signal_scorer.py)

```python
import anthropic
import json

client = anthropic.Anthropic(api_key=settings.ANTHROPIC_API_KEY)

SYSTEM_PROMPT = """You are a market intelligence AI for
Indian businesses. Analyze social media posts and return
ONLY valid JSON. No markdown. No explanation. Just JSON."""

def score_signal(post_title: str, post_content: str) -> dict:
    """
    Score a post for buyer intent using Claude Haiku.
    Returns None if signal is not relevant.
    """
    if not post_title and not post_content:
        return None

    text = f"Title: {post_title}\n\nContent: {post_content}"

    try:
        response = client.messages.create(
            model="claude-haiku-4-5-20251001",
            max_tokens=300,
            system=SYSTEM_PROMPT,
            messages=[{
                "role": "user",
                "content": f"""Analyze this internet post
for buyer intent from an Indian business perspective.

{text}

Return this exact JSON:
{{
  "signal_strength": <0-100 integer>,
  "signal_type": "<BUYING|SWITCHING|HELP_SEEKING|COMPLAINING|RESEARCHING|NOT_RELEVANT>",
  "urgency": "<HIGH|MEDIUM|LOW>",
  "language": "<en|hi|hinglish|ta|te>",
  "ai_summary": "<one sentence in English>",
  "reply_language": "<en|hi|hinglish|ta|te>"
}}

Signal strength guide:
90-100: Actively buying NOW — urgent
75-89:  Comparing/switching — strong intent
60-74:  Frustrated/seeking help
40-59:  Researching
0-39:   Not relevant (return NOT_RELEVANT)"""
            }]
        )

        text_response = response.content[0].text.strip()
        # Clean any markdown fences
        text_response = text_response.replace("```json","")
        text_response = text_response.replace("```","")

        scored = json.loads(text_response)

        # Skip irrelevant signals
        if (scored.get("signal_type") == "NOT_RELEVANT"
                or scored.get("signal_strength", 0) < 40):
            return None

        return scored

    except Exception as e:
        print(f"[Scorer] Error: {e}")
        return None
```

### 4.1 Auth Store (src/store/authStore.ts)

```typescript
import { create } from 'zustand'
import { persist, createJSONStorage } from 'zustand/middleware'

interface User {
  id: string
  email: string
  name: string
  phone: string
  plan: string
  onboarding_completed: boolean
}

interface AuthState {
  token: string | null
  user: User | null
  isAuthenticated: boolean
  setToken: (token: string) => void
  setUser: (user: User) => void
  setAuthenticated: (val: boolean) => void
  logout: () => void
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set) => ({
      token: null,
      user: null,
      isAuthenticated: false,
      setToken: (token) => set({ token }),
      setUser: (user) => set({ user }),
      setAuthenticated: (isAuthenticated) =>
        set({ isAuthenticated }),
      logout: () => set({
        token: null,
        user: null,
        isAuthenticated: false
      }),
    }),
    {
      name: 'nazar-auth',
      storage: createJSONStorage(() => localStorage),
    }
  )
)
```

### render.yaml (backend/)

```yaml
services:
  - type: web
    name: nazar-api
    env: python
    buildCommand: pip install -r requirements.txt
    startCommand: >
      alembic upgrade head &&
      uvicorn app.main:app --host 0.0.0.0
      --port $PORT --workers 2
    envVars:
      - key: APP_NAME
        value: Nazar
      - key: ENVIRONMENT
        value: production
      - key: TEST_MODE
        value: false
      - key: DATABASE_URL
        fromDatabase:
          name: nazar-db
          property: connectionString
      - key: REDIS_URL
        fromService:
          name: nazar-redis
          type: redis
          property: connectionString

  - type: worker
    name: nazar-celery-worker
    env: python
    buildCommand: pip install -r requirements.txt
    startCommand: >
      celery -A app.tasks.scrape_watch_zones
      worker --loglevel=info --concurrency=2

  - type: worker
    name: nazar-celery-beat
    env: python
    buildCommand: pip install -r requirements.txt
    startCommand: >
      celery -A app.tasks.scrape_watch_zones
      beat --loglevel=info

databases:
  - name: nazar-db
    databaseName: nazar
    user: nazar
    plan: free

```

### Render Setup Steps

```
1. Push backend/ to GitHub
   git init
   git add .
   git commit -m "Nazar backend"
   git remote add origin https://github.com/sameer/nazar-backend
   git push -u origin main

2. Go to render.com → New → Web Service
   Connect GitHub repo
   Select backend/ as root directory

3. Set environment variables in Render dashboard:
   SECRET_KEY      = [python -c "import secrets; print(secrets.token_hex(32))"]
   FRONTEND_URL    = https://the-nazar.com
   BACKEND_URL     = https://nazar-api.onrender.com
   ANTHROPIC_API_KEY = your_key
   SERPAPI_KEY     = your_key
   RAZORPAY_KEY_ID = your_key
   RAZORPAY_KEY_SECRET = your_key
   AISENSY_API_KEY = your_key
   RESEND_API_KEY  = your_key
   EMAIL_FROM      = signals@the-nazar.com

4. Add custom domain: api.the-nazar.com
   Add CNAME in GoDaddy DNS:
   api → nazar-api.onrender.com
```
